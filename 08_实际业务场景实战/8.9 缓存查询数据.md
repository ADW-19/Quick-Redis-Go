## 8.9 缓存查询数据

### 场景故事：内容平台的查询性能优化

**故事背景**：
某内容平台拥有大量的文章、视频等内容，用户频繁查询这些内容。随着用户量的增长，数据库查询压力越来越大，导致系统响应缓慢，用户体验差。技术团队需要优化查询性能，提高系统响应速度。

**业务需求**：
1. 提高内容查询速度，减少用户等待时间
2. 减轻数据库压力，提高系统稳定性
3. 支持复杂查询的缓存
4. 确保缓存数据的实时性
5. 支持批量查询优化

### 应用场景

- 热点数据缓存：热门文章、视频等高频访问内容
- 查询结果缓存：复杂查询的结果缓存
- 分页数据缓存：分页查询的结果缓存
- 复杂查询缓存：多条件组合查询的结果缓存

### 方案对比

**方案一：不使用缓存**
- **优点**：
  - 实现简单，无需额外缓存逻辑
  - 数据实时性最高
- **缺点**：
  - 数据库压力大，性能差
  - 响应速度慢，用户体验差
  - 无法应对高并发场景

**方案二：基于本地缓存**
- **优点**：
  - 实现简单，无网络开销
  - 访问速度快
- **缺点**：
  - 内存容量有限
  - 多服务器部署时缓存不一致
  - 服务器重启后缓存丢失
  - 无法共享缓存

**方案三：基于Redis的缓存**
- **优点**：
  - 高性能，支持高并发
  - 支持分布式部署，缓存一致性好
  - 支持过期时间，自动清理
  - 数据持久化选项（RDB/AOF）
  - 支持复杂查询的缓存
  - 支持批量查询优化
- **缺点**：
  - 需要额外的Redis服务器资源
  - 网络传输开销
  - 缓存数据可能存在短暂不一致
  - 实现复杂度较高

**选择理由**：
考虑到内容平台的查询性能需求和高并发场景，选择基于Redis的缓存查询方案。Redis的高性能特性能够显著提高查询速度，减轻数据库压力，同时支持复杂查询的缓存和批量查询优化。相比其他方案，Redis缓存查询系统在性能、可靠性和功能方面都具有明显优势，能够满足内容平台的业务需求。

### 代码实现

```go
package main

import (
	"context"
	"crypto/md5"
	"encoding/json"
	"fmt"
	"time"

	"github.com/go-redis/redis/v8"
)

// CacheQueryService 缓存查询服务
type CacheQueryService struct {
	rdb *redis.Client
}

// QueryConfig 查询配置
type QueryConfig struct {
	CacheKey        string
	ExpireTime      time.Duration
	RefreshTime     time.Duration // 提前刷新时间
	CacheNullResult bool          // 是否缓存空结果
	NullExpireTime  time.Duration // 空结果过期时间
}

// QueryResult 查询结果
type QueryResult struct {
	Data      interface{}
	FromCache bool
	CacheTime time.Time
}

// NewCacheQueryService 创建缓存查询服务
func NewCacheQueryService(rdb *redis.Client) *CacheQueryService {
	return &CacheQueryService{rdb: rdb}
}

// GenerateCacheKey 生成缓存Key
func (s *CacheQueryService) GenerateCacheKey(prefix string, params map[string]interface{}) string {
	// 将参数序列化为JSON
	paramJSON, _ := json.Marshal(params)
	
	// 使用MD5生成Key
	hash := md5.Sum(paramJSON)
	return fmt.Sprintf("%s:%x", prefix, hash)
}

// QueryWithCache 带缓存的查询
func (s *CacheQueryService) QueryWithCache(ctx context.Context, config QueryConfig, queryFunc func() (interface{}, error)) (*QueryResult, error) {
	// 1. 查询缓存
	cachedData, err := s.rdb.Get(ctx, config.CacheKey).Result()
	if err == nil {
		// 缓存命中
		var data interface{}
		if err := json.Unmarshal([]byte(cachedData), &data); err != nil {
			return nil, err
		}
		
		// 检查是否需要提前刷新（异步刷新）
		ttl, _ := s.rdb.TTL(ctx, config.CacheKey).Result()
		if ttl > 0 && ttl < config.RefreshTime {
			// 异步刷新缓存
			go s.refreshCache(context.Background(), config, queryFunc)
		}
		
		return &QueryResult{
			Data:      data,
			FromCache: true,
			CacheTime: time.Now(),
		}, nil
	}
	
	if err != redis.Nil {
		return nil, err
	}
	
	// 2. 缓存未命中，查询数据源
	data, err := queryFunc()
	if err != nil {
		return nil, err
	}
	
	// 3. 写入缓存
	if data != nil || config.CacheNullResult {
		jsonData, _ := json.Marshal(data)
		
		expireTime := config.ExpireTime
		if data == nil {
			expireTime = config.NullExpireTime
		}
		
		err = s.rdb.Set(ctx, config.CacheKey, jsonData, expireTime).Err()
		if err != nil {
			// 缓存写入失败不影响返回结果
			fmt.Printf("写入缓存失败: %v\n", err)
		}
	}
	
	return &QueryResult{
		Data:      data,
		FromCache: false,
		CacheTime: time.Now(),
	}, nil
}

// refreshCache 异步刷新缓存
func (s *CacheQueryService) refreshCache(ctx context.Context, config QueryConfig, queryFunc func() (interface{}, error)) {
	data, err := queryFunc()
	if err != nil {
		fmt.Printf("异步刷新缓存失败: %v\n", err)
		return
	}
	
	jsonData, _ := json.Marshal(data)
	err = s.rdb.Set(ctx, config.CacheKey, jsonData, config.ExpireTime).Err()
	if err != nil {
		fmt.Printf("异步刷新缓存写入失败: %v\n", err)
		return
	}
	
	fmt.Printf("异步刷新缓存成功: %s\n", config.CacheKey)
}

// BatchQueryWithCache 批量查询（使用Pipeline优化）
func (s *CacheQueryService) BatchQueryWithCache(ctx context.Context, configs []QueryConfig, queryFuncs []func() (interface{}, error)) ([]*QueryResult, error) {
	if len(configs) != len(queryFuncs) {
		return nil, fmt.Errorf("配置和查询函数数量不匹配")
	}
	
	results := make([]*QueryResult, len(configs))
	
	// 1. 批量查询缓存
	pipe := s.rdb.Pipeline()
	cmds := make([]*redis.StringCmd, len(configs))
	
	for i, config := range configs {
		cmds[i] = pipe.Get(ctx, config.CacheKey)
	}
	
	_, err := pipe.Exec(ctx)
	if err != nil && err != redis.Nil {
		return nil, err
	}
	
	// 2. 处理缓存结果
	missedIndices := []int{}
	for i, cmd := range cmds {
		cachedData, err := cmd.Result()
		if err == nil {
			// 缓存命中
			var data interface{}
			if err := json.Unmarshal([]byte(cachedData), &data); err != nil {
				return nil, err
			}
			
			results[i] = &QueryResult{
				Data:      data,
				FromCache: true,
				CacheTime: time.Now(),
			}
		} else if err == redis.Nil {
			// 缓存未命中
			missedIndices = append(missedIndices, i)
		} else {
			return nil, err
		}
	}
	
	// 3. 查询未命中的数据
	if len(missedIndices) > 0 {
		pipe = s.rdb.Pipeline()
		
		for _, idx := range missedIndices {
			data, err := queryFuncs[idx]()
			if err != nil {
				return nil, err
			}
			
			results[idx] = &QueryResult{
				Data:      data,
				FromCache: false,
				CacheTime: time.Now(),
			}
			
			// 写入缓存
			if data != nil || configs[idx].CacheNullResult {
				jsonData, _ := json.Marshal(data)
				
				expireTime := configs[idx].ExpireTime
				if data == nil {
					expireTime = configs[idx].NullExpireTime
				}
				
				pipe.Set(ctx, configs[idx].CacheKey, jsonData, expireTime)
			}
		}
		
		_, err = pipe.Exec(ctx)
		if err != nil {
			fmt.Printf("批量写入缓存失败: %v\n", err)
		}
	}
	
	return results, nil
}

// DeleteCache 删除缓存
func (s *CacheQueryService) DeleteCache(ctx context.Context, cacheKey string) error {
	return s.rdb.Del(ctx, cacheKey).Err()
}

// DeleteCachePattern 按模式删除缓存
func (s *CacheQueryService) DeleteCachePattern(ctx context.Context, pattern string) error {
	iter := s.rdb.Scan(ctx, 0, pattern, 0).Iterator()
	
	for iter.Next(ctx) {
		key := iter.Val()
		err := s.rdb.Del(ctx, key).Err()
		if err != nil {
			fmt.Printf("删除缓存失败 %s: %v\n", key, err)
		}
	}
	
	return iter.Err()
}

// 使用示例
func main() {
	ctx := context.Background()
	
	// 创建Redis客户端
	rdb := redis.NewClient(&redis.Options{
		Addr: "localhost:6379",
	})
	
	// 创建缓存查询服务
	service := NewCacheQueryService(rdb)
	
	// 示例1：单个查询
	params := map[string]interface{}{
		"user_id": 12345,
		"type":    "profile",
	}
	
	cacheKey := service.GenerateCacheKey("user", params)
	config := QueryConfig{
		CacheKey:        cacheKey,
		ExpireTime:      5 * time.Minute,
		RefreshTime:     1 * time.Minute,
		CacheNullResult: true,
		NullExpireTime:  1 * time.Minute,
	}
	
	// 模拟查询函数
	queryFunc := func() (interface{}, error) {
		// 实际项目中这里会查询数据库
		fmt.Println("查询数据库...")
		return map[string]interface{}{
			"user_id":  12345,
			"username": "张三",
			"email":    "zhangsan@example.com",
		}, nil
	}
	
	// 第一次查询（缓存未命中）
	result1, err := service.QueryWithCache(ctx, config, queryFunc)
	if err != nil {
		fmt.Printf("查询失败: %v\n", err)
		return
	}
	fmt.Printf("第一次查询结果: %+v, 来自缓存: %v\n", result1.Data, result1.FromCache)
	
	// 第二次查询（缓存命中）
	result2, err := service.QueryWithCache(ctx, config, queryFunc)
	if err != nil {
		fmt.Printf("查询失败: %v\n", err)
		return
	}
	fmt.Printf("第二次查询结果: %+v, 来自缓存: %v\n", result2.Data, result2.FromCache)
	
	// 示例2：批量查询
	fmt.Println("\n=== 批量查询示例 ===")
	
	batchConfigs := []QueryConfig{
		{
			CacheKey:   service.GenerateCacheKey("product", map[string]interface{}{"id": 1}),
			ExpireTime: 10 * time.Minute,
		},
		{
			CacheKey:   service.GenerateCacheKey("product", map[string]interface{}{"id": 2}),
			ExpireTime: 10 * time.Minute,
		},
		{
			CacheKey:   service.GenerateCacheKey("product", map[string]interface{}{"id": 3}),
			ExpireTime: 10 * time.Minute,
		},
	}
	
	batchQueryFuncs := []func() (interface{}, error){
		func() (interface{}, error) {
			fmt.Println("查询商品1...")
			return map[string]interface{}{"id": 1, "name": "商品1", "price": 100}, nil
		},
		func() (interface{}, error) {
			fmt.Println("查询商品2...")
			return map[string]interface{}{"id": 2, "name": "商品2", "price": 200}, nil
		},
		func() (interface{}, error) {
			fmt.Println("查询商品3...")
			return map[string]interface{}{"id": 3, "name": "商品3", "price": 300}, nil
		},
	}
	
	batchResults, err := service.BatchQueryWithCache(ctx, batchConfigs, batchQueryFuncs)
	if err != nil {
		fmt.Printf("批量查询失败: %v\n", err)
		return
	}
	
	for i, result := range batchResults {
		fmt.Printf("批量查询结果 %d: %+v, 来自缓存: %v\n", i+1, result.Data, result.FromCache)
	}
}
```
