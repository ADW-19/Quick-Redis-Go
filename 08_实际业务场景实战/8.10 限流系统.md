## 8.10 限流系统

### 场景故事：API接口的保护需求

**故事背景**：
某API服务提供商发现部分用户滥用API接口，发送大量请求，导致系统负载过高，影响其他用户的正常使用。同时，恶意攻击者尝试通过暴力请求来攻击系统，造成服务不稳定。技术团队需要实现一个限流系统来保护API接口。

**业务需求**：
1. 限制API接口的调用频率，防止滥用
2. 保护系统免受暴力攻击
3. 提供公平的资源分配
4. 支持不同维度的限流（IP、用户、接口）
5. 实现平滑的流量控制

### 应用场景

- API接口限流：限制API的调用频率
- 用户操作频率限制：限制用户的操作次数
- 防刷保护：防止恶意刷取
- 资源访问控制：控制对有限资源的访问

### 方案对比

**方案一：基于计数器的限流**
- **优点**：
  - 实现简单，逻辑清晰
  - 内存占用小
- **缺点**：
  - 存在边界效应，可能在窗口切换时允许突发流量
  - 无法实现平滑的流量控制
  - 精度较低

**方案二：基于令牌桶的限流**
- **优点**：
  - 实现平滑的流量控制
  - 支持突发流量
  - 精度高
- **缺点**：
  - 实现复杂度较高
  - 需要维护令牌桶状态
  - 内存占用较大

**方案三：基于Redis的限流**
- **优点**：
  - 支持分布式部署
  - 高性能，支持高并发
  - 实现简单，API友好
  - 支持多种限流算法
  - 支持不同维度的限流
- **缺点**：
  - 需要额外的Redis服务器资源
  - 网络传输开销
  - 依赖Redis的稳定性

**选择理由**：
考虑到API服务的高并发需求和分布式部署场景，选择基于Redis的限流系统方案。Redis的高性能特性能够应对大量的限流请求，支持分布式部署，同时提供多种限流算法，满足不同场景的需求。相比其他方案，Redis限流系统在性能、可靠性和功能方面都具有明显优势，能够有效保护API接口。

### 代码实现

```go
package main

import (
	"context"
	"fmt"
	"time"

	"github.com/go-redis/redis/v8"
)

// RateLimiter 限流器接口
type RateLimiter interface {
	Allow(ctx context.Context, key string) (bool, error)
}

// SlidingWindowLimiter 滑动窗口限流器
type SlidingWindowLimiter struct {
	rdb        *redis.Client
	windowSize time.Duration
	maxCount   int64
}

// NewSlidingWindowLimiter 创建滑动窗口限流器
func NewSlidingWindowLimiter(rdb *redis.Client, windowSize time.Duration, maxCount int64) *SlidingWindowLimiter {
	return &SlidingWindowLimiter{
		rdb:        rdb,
		windowSize: windowSize,
		maxCount:   maxCount,
	}
}

// Allow 检查是否允许通过
func (l *SlidingWindowLimiter) Allow(ctx context.Context, key string) (bool, error) {
	now := time.Now().UnixNano() / 1e6 // 毫秒时间戳
	windowStart := now - l.windowSize.Milliseconds()
	
	pipe := l.rdb.Pipeline()
	
	// 移除窗口外的请求记录
	pipe.ZRemRangeByScore(ctx, key, "0", fmt.Sprintf("%d", windowStart))
	
	// 获取当前窗口内的请求数
	countCmd := pipe.ZCard(ctx, key)
	
	// 添加当前请求
	pipe.ZAdd(ctx, key, &redis.Z{
		Score:  float64(now),
		Member: now,
	})
	
	// 设置过期时间
	pipe.Expire(ctx, key, l.windowSize)
	
	_, err := pipe.Exec(ctx)
	if err != nil {
		return false, err
	}
	
	count := countCmd.Val()
	return count < l.maxCount, nil
}

// TokenBucketLimiter 令牌桶限流器
type TokenBucketLimiter struct {
	rdb         *redis.Client
	bucketSize  int64  // 桶容量
	refillRate  int64  // 每秒填充令牌数
}

// NewTokenBucketLimiter 创建令牌桶限流器
func NewTokenBucketLimiter(rdb *redis.Client, bucketSize, refillRate int64) *TokenBucketLimiter {
	return &TokenBucketLimiter{
		rdb:        rdb,
		bucketSize: bucketSize,
		refillRate: refillRate,
	}
}

// Allow 检查是否允许通过
func (l *TokenBucketLimiter) Allow(ctx context.Context, key string) (bool, error) {
	// Lua脚本实现令牌桶算法
	script := `
		local bucketKey = KEYS[1]
		local bucketSize = tonumber(ARGV[1])
		local refillRate = tonumber(ARGV[2])
		local now = tonumber(ARGV[3])
		
		-- 获取当前令牌数和上次更新时间
		local bucketData = redis.call('HMGET', bucketKey, 'tokens', 'lastRefill')
		local tokens = tonumber(bucketData[1])
		local lastRefill = tonumber(bucketData[2])
		
		-- 初始化桶
		if tokens == nil then
			tokens = bucketSize
			lastRefill = now
		end
		
		-- 计算需要填充的令牌数
		local elapsed = math.max(0, now - lastRefill)
		local refillTokens = math.floor(elapsed * refillRate / 1000)
		
		-- 填充令牌
		tokens = math.min(bucketSize, tokens + refillTokens)
		lastRefill = now
		
		-- 尝试消费令牌
		if tokens >= 1 then
			tokens = tokens - 1
			redis.call('HMSET', bucketKey, 'tokens', tokens, 'lastRefill', lastRefill)
			redis.call('EXPIRE', bucketKey, 60)
			return 1
		else
			redis.call('HMSET', bucketKey, 'tokens', tokens, 'lastRefill', lastRefill)
			redis.call('EXPIRE', bucketKey, 60)
			return 0
		end
	`
	
	now := time.Now().UnixNano() / 1e6 // 毫秒时间戳
	
	result, err := l.rdb.Eval(ctx, script, []string{key}, 
		l.bucketSize, l.refillRate, now).Result()
	if err != nil {
		return false, err
	}
	
	return result.(int64) == 1, nil
}

// RateLimitService 限流服务
type RateLimitService struct {
	rdb *redis.Client
}

// NewRateLimitService 创建限流服务
func NewRateLimitService(rdb *redis.Client) *RateLimitService {
	return &RateLimitService{rdb: rdb}
}

// LimitByIP IP限流
func (s *RateLimitService) LimitByIP(ctx context.Context, ip string) (bool, error) {
	limiter := NewSlidingWindowLimiter(s.rdb, time.Minute, 100)
	key := fmt.Sprintf("ratelimit:ip:%s", ip)
	return limiter.Allow(ctx, key)
}

// LimitByUser 用户限流
func (s *RateLimitService) LimitByUser(ctx context.Context, userID string) (bool, error) {
	limiter := NewSlidingWindowLimiter(s.rdb, time.Minute, 60)
	key := fmt.Sprintf("ratelimit:user:%s", userID)
	return limiter.Allow(ctx, key)
}

// LimitByAPI API限流
func (s *RateLimitService) LimitByAPI(ctx context.Context, api string) (bool, error) {
	limiter := NewTokenBucketLimiter(s.rdb, 100, 10)
	key := fmt.Sprintf("ratelimit:api:%s", api)
	return limiter.Allow(ctx, key)
}

// 使用示例
func main() {
	ctx := context.Background()
	
	// 创建Redis客户端
	rdb := redis.NewClient(&redis.Options{
		Addr: "localhost:6379",
	})
	
	// 创建限流服务
	service := NewRateLimitService(rdb)
	
	// 测试IP限流
	ip := "192.168.1.100"
	fmt.Println("=== IP限流测试 ===")
	for i := 0; i < 105; i++ {
		allowed, err := service.LimitByIP(ctx, ip)
		if err != nil {
			fmt.Printf("限流检查失败: %v\n", err)
			continue
		}
		if i < 10 || i >= 100 {
			fmt.Printf("请求 %d: %v\n", i+1, allowed)
		}
	}
	
	// 测试API限流
	api := "/api/v1/users"
	fmt.Println("\n=== API限流测试 ===")
	for i := 0; i < 15; i++ {
		allowed, err := service.LimitByAPI(ctx, api)
		if err != nil {
			fmt.Printf("限流检查失败: %v\n", err)
			continue
		}
		fmt.Printf("请求 %d: %v\n", i+1, allowed)
		time.Sleep(100 * time.Millisecond)
	}
}
```
