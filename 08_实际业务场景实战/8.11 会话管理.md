## 8.11 会话管理

### 场景故事：分布式系统的会话共享需求

**故事背景**：
某电商平台采用分布式部署架构，用户在不同服务器间切换时，会话数据丢失，导致用户需要重新登录，购物车数据丢失等问题。技术团队需要实现一个分布式会话管理系统，确保用户在不同服务器间无缝切换。

**业务需求**：
1. 实现分布式会话共享
2. 确保会话数据的安全性
3. 支持会话过期管理
4. 提供良好的用户体验
5. 支持高并发场景

### 应用场景

- 用户登录状态管理：保持用户登录状态
- 购物车数据管理：保存用户购物车信息
- 临时会话数据存储：存储临时会话信息
- 分布式系统的会话共享：确保不同服务器间的会话一致性

### 方案对比

**方案一：基于Cookie的会话管理**
- **优点**：
  - 实现简单，无需服务器存储
  - 无服务器负担
- **缺点**：
  - 存储容量有限（通常4KB）
  - 安全性差，容易被篡改
  - 无法存储敏感信息
  - 依赖客户端Cookie设置

**方案二：基于数据库的会话管理**
- **优点**：
  - 数据持久化，可靠性高
  - 支持复杂查询
  - 安全性较好
- **缺点**：
  - 性能差，数据库成为瓶颈
  - 并发能力有限
  - 响应速度慢
  - 难以应对高并发场景

**方案三：基于Redis的会话管理**
- **优点**：
  - 高性能，支持高并发
  - 支持分布式部署，会话共享
  - 支持过期时间，自动清理
  - 数据持久化选项（RDB/AOF）
  - 安全性好，支持加密
  - 响应速度快
- **缺点**：
  - 需要额外的Redis服务器资源
  - 网络传输开销
  - 依赖Redis的稳定性

**选择理由**：
考虑到电商平台的分布式部署架构和高并发需求，选择基于Redis的会话管理方案。Redis的高性能特性能够应对大量的会话操作，支持分布式部署确保会话共享，同时提供过期时间管理和数据持久化选项，确保会话数据的安全性和可靠性。相比其他方案，Redis会话管理系统在性能、可靠性和功能方面都具有明显优势，能够满足电商平台的业务需求。

### 代码实现

```go
package main

import (
	"context"
	"encoding/json"
	"fmt"
	"time"

	"github.com/go-redis/redis/v8"
)

// Session 会话结构
type Session struct {
	ID        string                 `json:"id"`
	UserID    string                 `json:"user_id"`
	Data      map[string]interface{} `json:"data"`
	CreatedAt time.Time              `json:"created_at"`
	UpdatedAt time.Time              `json:"updated_at"`
}

// SessionService 会话服务
type SessionService struct {
	rdb          *redis.Client
	sessionTTL   time.Duration
}

// NewSessionService 创建会话服务
func NewSessionService(rdb *redis.Client, sessionTTL time.Duration) *SessionService {
	return &SessionService{
		rdb:        rdb,
		sessionTTL: sessionTTL,
	}
}

// CreateSession 创建会话
func (s *SessionService) CreateSession(ctx context.Context, userID string, data map[string]interface{}) (string, error) {
	sessionID := fmt.Sprintf("session:%d", time.Now().UnixNano())
	session := Session{
		ID:        sessionID,
		UserID:    userID,
		Data:      data,
		CreatedAt: time.Now(),
		UpdatedAt: time.Now(),
	}
	
	dataBytes, err := json.Marshal(session)
	if err != nil {
		return "", err
	}
	
	err = s.rdb.Set(ctx, sessionID, dataBytes, s.sessionTTL).Err()
	if err != nil {
		return "", err
	}
	
	return sessionID, nil
}

// GetSession 获取会话
func (s *SessionService) GetSession(ctx context.Context, sessionID string) (*Session, error) {
	dataBytes, err := s.rdb.Get(ctx, sessionID).Bytes()
	if err != nil {
		if err == redis.Nil {
			return nil, nil // 会话不存在
		}
		return nil, err
	}
	
	var session Session
	err = json.Unmarshal(dataBytes, &session)
	if err != nil {
		return nil, err
	}
	
	// 延长会话过期时间
	s.rdb.Expire(ctx, sessionID, s.sessionTTL)
	
	return &session, nil
}

// UpdateSession 更新会话
func (s *SessionService) UpdateSession(ctx context.Context, sessionID string, data map[string]interface{}) error {
	session, err := s.GetSession(ctx, sessionID)
	if err != nil {
		return err
	}
	
	if session == nil {
		return fmt.Errorf("session not found")
	}
	
	// 更新会话数据
	for k, v := range data {
		session.Data[k] = v
	}
	session.UpdatedAt = time.Now()
	
	dataBytes, err := json.Marshal(session)
	if err != nil {
		return err
	}
	
	return s.rdb.Set(ctx, sessionID, dataBytes, s.sessionTTL).Err()
}

// DeleteSession 删除会话
func (s *SessionService) DeleteSession(ctx context.Context, sessionID string) error {
	return s.rdb.Del(ctx, sessionID).Err()
}

// GetUserSessions 获取用户的所有会话
func (s *SessionService) GetUserSessions(ctx context.Context, userID string) ([]*Session, error) {
	// 这里简化处理，实际项目中可能需要使用Set存储用户会话ID
	// 或者使用Redis的SCAN命令查找用户的会话
	// 这里仅作为示例
	return []*Session{}, nil
}

// 使用示例
func main() {
	ctx := context.Background()
	
	// 创建Redis客户端
	rdb := redis.NewClient(&redis.Options{
		Addr: "localhost:6379",
	})
	
	// 创建会话服务
	service := NewSessionService(rdb, 24*time.Hour)
	
	// 创建会话
	userID := "user123"
	sessionData := map[string]interface{}{
		"username": "testuser",
		"email":    "test@example.com",
		"cart":     []string{"item1", "item2"},
	}
	
	sessionID, err := service.CreateSession(ctx, userID, sessionData)
	if err != nil {
		fmt.Printf("创建会话失败: %v\n", err)
		return
	}
	fmt.Printf("创建会话成功，会话ID: %s\n", sessionID)
	
	// 获取会话
	session, err := service.GetSession(ctx, sessionID)
	if err != nil {
		fmt.Printf("获取会话失败: %v\n", err)
		return
	}
	if session == nil {
		fmt.Println("会话不存在")
		return
	}
	fmt.Printf("获取会话成功: %+v\n", session)
	
	// 更新会话
	updateData := map[string]interface{}{
		"cart": []string{"item1", "item2", "item3"},
	}
	err = service.UpdateSession(ctx, sessionID, updateData)
	if err != nil {
		fmt.Printf("更新会话失败: %v\n", err)
		return
	}
	fmt.Println("更新会话成功")
	
	// 再次获取会话
	updatedSession, err := service.GetSession(ctx, sessionID)
	if err != nil {
		fmt.Printf("获取会话失败: %v\n", err)
		return
	}
	fmt.Printf("更新后会话: %+v\n", updatedSession)
	
	// 删除会话
	err = service.DeleteSession(ctx, sessionID)
	if err != nil {
		fmt.Printf("删除会话失败: %v\n", err)
		return
	}
	fmt.Println("删除会话成功")
	
	// 验证会话是否删除
	deletedSession, err := service.GetSession(ctx, sessionID)
	if err != nil {
		fmt.Printf("获取会话失败: %v\n", err)
		return
	}
	if deletedSession == nil {
		fmt.Println("会话已成功删除")
	} else {
		fmt.Println("会话删除失败")
	}
}
```
