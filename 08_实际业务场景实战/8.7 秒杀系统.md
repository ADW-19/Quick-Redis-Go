## 8.7 秒杀系统

### 场景故事：电商秒杀活动的技术挑战

**故事背景**：
某电商平台计划举办一场限时秒杀活动，推出限量特价商品。技术团队担心系统无法应对瞬时的高并发请求，可能导致系统崩溃、超卖、订单创建失败等问题。历史数据显示，秒杀活动开始时，系统会面临平时10倍以上的流量冲击。

**业务需求**：
1. 处理高并发请求，确保系统稳定
2. 防止超卖，确保库存准确性
3. 保证公平性，避免部分用户无法参与
4. 提供良好的用户体验，快速响应
5. 支持异步处理，减轻系统压力

### 应用场景

- 电商秒杀活动：限时特价商品抢购
- 抢购活动：限量商品抢购
- 限时优惠：特定时间段的优惠活动
- 门票抢购：热门赛事、演出门票

### 方案对比

**方案一：基于数据库的秒杀系统**
- **优点**：
  - 实现简单，利用现有数据库
  - 数据持久化，可靠性高
  - 事务支持，数据一致性好
- **缺点**：
  - 性能差，数据库成为瓶颈
  - 并发能力有限，容易产生锁竞争
  - 无法应对高并发场景
  - 容易出现超卖问题

**方案二：基于内存的秒杀系统**
- **优点**：
  - 性能高，内存操作
  - 实时性好，无延迟
- **缺点**：
  - 数据易丢失，服务器重启后数据清零
  - 不支持分布式环境
  - 内存容量有限
  - 无法实现跨服务器同步

**方案三：基于Redis的秒杀系统**
- **优点**：
  - 高性能，支持高并发
  - 原子操作，防止超卖
  - 支持分布式部署
  - 内存操作，响应速度快
  - 支持限流，保护系统
  - 支持异步处理，减轻系统压力
- **缺点**：
  - 需要额外的Redis服务器资源
  - 网络传输开销
  - 数据可能存在短暂不一致（在主从复制场景下）
  - 实现复杂度较高

**选择理由**：
考虑到电商秒杀活动的高并发需求和防止超卖的要求，选择基于Redis的秒杀系统方案。Redis的原子操作能够保证库存扣减的准确性，高性能特性能够应对瞬时的流量冲击，同时支持限流和异步处理，有效保护系统。相比其他方案，Redis秒杀系统在性能、可靠性和功能方面都具有明显优势，能够满足秒杀活动的业务需求。

### 代码实现

```go
package main

import (
	"context"
	"fmt"
	"time"

	"github.com/go-redis/redis/v8"
)

// SeckillService 秒杀服务
type SeckillService struct {
	rdb *redis.Client
}

// SeckillConfig 秒杀配置
type SeckillConfig struct {
	StockKey        string        // 库存键
	OrderQueueKey   string        // 订单队列键
	UserLimitKey    string        // 用户限购键
	RateLimitKey    string        // 限流键
	RateLimitCount  int           // 限流次数
	RateLimitWindow time.Duration // 限流时间窗口
}

// NewSeckillService 创建秒杀服务
func NewSeckillService(rdb *redis.Client) *SeckillService {
	return &SeckillService{rdb: rdb}
}

// InitStock 初始化库存
func (s *SeckillService) InitStock(ctx context.Context, productID string, stock int) error {
	stockKey := fmt.Sprintf("seckill:stock:%s", productID)
	return s.rdb.Set(ctx, stockKey, stock, 0).Err()
}

// CheckRateLimit 检查限流
func (s *SeckillService) CheckRateLimit(ctx context.Context, userID string) (bool, error) {
	config := SeckillConfig{
		RateLimitCount:  1,
		RateLimitWindow: time.Second,
	}
	
	rateKey := fmt.Sprintf("seckill:rate:%s", userID)
	count, err := s.rdb.Incr(ctx, rateKey).Result()
	if err != nil {
		return false, err
	}
	
	if count == 1 {
		s.rdb.Expire(ctx, rateKey, config.RateLimitWindow)
	}
	
	if count > int64(config.RateLimitCount) {
		return false, nil
	}
	
	return true, nil
}

// CheckUserLimit 检查用户限购
func (s *SeckillService) CheckUserLimit(ctx context.Context, userID string, productID string) (bool, error) {
	limitKey := fmt.Sprintf("seckill:limit:%s:%s", productID, userID)
	exists, err := s.rdb.Exists(ctx, limitKey).Result()
	if err != nil {
		return false, err
	}
	
	return exists == 0, nil
}

// DeductStock 扣减库存（使用Lua脚本保证原子性）
func (s *SeckillService) DeductStock(ctx context.Context, productID string) (bool, error) {
	stockKey := fmt.Sprintf("seckill:stock:%s", productID)
	
	// Lua脚本：原子性扣减库存
	script := `
		local stock = tonumber(redis.call('get', KEYS[1]))
		if stock == nil then
			return -1
		end
		if stock <= 0 then
			return 0
		end
		redis.call('decr', KEYS[1])
		return 1
	`
	
	result, err := s.rdb.Eval(ctx, script, []string{stockKey}).Result()
	if err != nil {
		return false, err
	}
	
	switch result.(int64) {
	case -1:
		return false, fmt.Errorf("库存未初始化")
	case 0:
		return false, nil // 库存不足
	case 1:
		return true, nil  // 扣减成功
	default:
		return false, fmt.Errorf("未知错误")
	}
}

// Seckill 秒杀核心逻辑
func (s *SeckillService) Seckill(ctx context.Context, userID string, productID string) (string, error) {
	// 1. 限流检查
	allowed, err := s.CheckRateLimit(ctx, userID)
	if err != nil {
		return "", fmt.Errorf("限流检查失败: %w", err)
	}
	if !allowed {
		return "", fmt.Errorf("请求过于频繁，请稍后再试")
	}
	
	// 2. 用户限购检查
	canBuy, err := s.CheckUserLimit(ctx, userID, productID)
	if err != nil {
		return "", fmt.Errorf("限购检查失败: %w", err)
	}
	if !canBuy {
		return "", fmt.Errorf("您已经参与过该商品的秒杀")
	}
	
	// 3. 扣减库存
	success, err := s.DeductStock(ctx, productID)
	if err != nil {
		return "", fmt.Errorf("扣减库存失败: %w", err)
	}
	if !success {
		return "", fmt.Errorf("商品库存不足")
	}
	
	// 4. 标记用户已购买
	limitKey := fmt.Sprintf("seckill:limit:%s:%s", productID, userID)
	s.rdb.Set(ctx, limitKey, 1, 24*time.Hour)
	
	// 5. 生成订单Token
	orderToken := fmt.Sprintf("order:%s:%s:%d", productID, userID, time.Now().Unix())
	
	// 6. 异步创建订单（发送到消息队列）
	orderQueueKey := fmt.Sprintf("seckill:queue:%s", productID)
	orderInfo := fmt.Sprintf("%s|%s|%s", orderToken, userID, productID)
	s.rdb.RPush(ctx, orderQueueKey, orderInfo)
	
	return orderToken, nil
}

// GetStock 获取库存
func (s *SeckillService) GetStock(ctx context.Context, productID string) (int64, error) {
	stockKey := fmt.Sprintf("seckill:stock:%s", productID)
	return s.rdb.Get(ctx, stockKey).Int64()
}

// ProcessOrders 处理订单（消费者）
func (s *SeckillService) ProcessOrders(ctx context.Context, productID string) {
	orderQueueKey := fmt.Sprintf("seckill:queue:%s", productID)
	
	for {
		// 阻塞式获取订单
		result, err := s.rdb.BLPop(ctx, 0, orderQueueKey).Result()
		if err != nil {
			fmt.Printf("获取订单失败: %v\n", err)
			continue
		}
		
		orderInfo := result[1]
		// 解析订单信息
		// 实际项目中这里会调用订单服务创建订单
		fmt.Printf("处理订单: %s\n", orderInfo)
		
		// 模拟订单处理
		time.Sleep(100 * time.Millisecond)
		fmt.Printf("订单处理完成: %s\n", orderInfo)
	}
}

// 使用示例
func main() {
	ctx := context.Background()
	
	// 创建Redis客户端
	rdb := redis.NewClient(&redis.Options{
		Addr: "localhost:6379",
	})
	
	// 创建秒杀服务
	service := NewSeckillService(rdb)
	
	productID := "product_001"
	
	// 初始化库存
	err := service.InitStock(ctx, productID, 100)
	if err != nil {
		fmt.Printf("初始化库存失败: %v\n", err)
		return
	}
	
	// 启动订单处理器
	go service.ProcessOrders(ctx, productID)
	
	// 模拟多个用户秒杀
	for i := 1; i <= 10; i++ {
		userID := fmt.Sprintf("user_%d", i)
		
		// 模拟并发秒杀
		go func(uid string) {
			orderToken, err := service.Seckill(ctx, uid, productID)
			if err != nil {
				fmt.Printf("用户 %s 秒杀失败: %v\n", uid, err)
				return
			}
			fmt.Printf("用户 %s 秒杀成功，订单Token: %s\n", uid, orderToken)
		}(userID)
	}
	
	// 等待处理完成
	time.Sleep(5 * time.Second)
	
	// 查看剩余库存
	stock, _ := service.GetStock(ctx, productID)
	fmt.Printf("剩余库存: %d\n", stock)
}
```
