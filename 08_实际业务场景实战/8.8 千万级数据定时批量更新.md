## 8.8 千万级数据定时批量更新

### 场景故事：电商平台的数据分析需求

**故事背景**：
某电商平台拥有千万级的商品数据，需要定期更新商品信息、计算销售统计数据、同步用户信息等。传统的单线程处理方式耗时过长，影响系统性能，无法在规定时间内完成任务。技术团队需要设计一个高效的批量更新系统，能够处理千万级数据的定时更新任务。

**业务需求**：
1. 高效处理千万级数据的批量更新
2. 支持定时任务调度
3. 提供任务进度监控
4. 支持失败重试机制
5. 减少对生产系统的影响

### 应用场景

- 商品信息批量更新：更新商品价格、库存等信息
- 用户数据批量同步：同步用户信息到其他系统
- 统计数据批量计算：计算销售统计、用户行为分析等
- 缓存数据批量刷新：刷新缓存数据，确保数据一致性

### 方案对比

**方案一：单线程批量更新**
- **优点**：
  - 实现简单，逻辑清晰
  - 不需要复杂的并发控制
- **缺点**：
  - 处理速度慢，耗时过长
  - 无法充分利用系统资源
  - 容易超时
  - 影响系统性能

**方案二：多线程批量更新**
- **优点**：
  - 处理速度快，充分利用系统资源
  - 支持并发处理
- **缺点**：
  - 实现复杂，需要并发控制
  - 可能产生线程竞争
  - 内存消耗大
  - 难以监控任务进度

**方案三：基于Redis的分布式批量更新**
- **优点**：
  - 高性能，支持分布式处理
  - 任务分片，并行处理
  - 进度监控，实时了解任务状态
  - 支持失败重试
  - 减少对生产系统的影响
  - 可扩展性好
- **缺点**：
  - 实现复杂度较高
  - 需要额外的Redis服务器资源
  - 网络传输开销
  - 依赖Redis的稳定性

**选择理由**：
考虑到电商平台的千万级数据处理需求和系统性能要求，选择基于Redis的分布式批量更新方案。Redis的高性能特性和分布式支持能够实现高效的数据处理，任务分片和并行处理能够充分利用系统资源，同时提供实时的进度监控和失败重试机制，确保任务能够在规定时间内完成。相比其他方案，Redis分布式批量更新系统在性能、可靠性和可扩展性方面都具有明显优势。

package main

import (
	"context"
	"fmt"
	"strconv"
	"strings"
	"sync"
	"time"

	"github.com/go-redis/redis/v8"
)

// BatchUpdateService 批量更新服务
type BatchUpdateService struct {
	rdb        *redis.Client
	workerCount int
	batchSize   int
}

// TaskConfig 任务配置
type TaskConfig struct {
	TaskID      string
	TotalCount  int64
	ShardSize   int64
	StartTime   time.Time
	ExpireTime  time.Duration
}

// ShardStatus 分片状态
type ShardStatus struct {
	ShardID    string
	StartID    int64
	EndID      int64
	Status     string // pending, processing, completed, failed
	Progress   int64
	UpdateTime time.Time
}

// NewBatchUpdateService 创建批量更新服务
func NewBatchUpdateService(rdb *redis.Client, workerCount, batchSize int) *BatchUpdateService {
	return &BatchUpdateService{
		rdb:         rdb,
		workerCount: workerCount,
		batchSize:   batchSize,
	}
}

// CreateTask 创建批量更新任务
func (s *BatchUpdateService) CreateTask(ctx context.Context, taskID string, totalCount int64) (*TaskConfig, error) {
	shardSize := int64(100000) // 每个分片10万条数据
	shardCount := (totalCount + shardSize - 1) / shardSize
	
	config := &TaskConfig{
		TaskID:     taskID,
		TotalCount: totalCount,
		ShardSize:  shardSize,
		StartTime:  time.Now(),
		ExpireTime: 24 * time.Hour,
	}
	
	// 存储任务配置
	configKey := fmt.Sprintf("batch:task:%s:config", taskID)
	s.rdb.HSet(ctx, configKey, map[string]interface{}{
		"total_count": totalCount,
		"shard_size":  shardSize,
		"shard_count": shardCount,
		"start_time":  config.StartTime.Format(time.RFC3339),
		"status":      "created",
	})
	s.rdb.Expire(ctx, configKey, config.ExpireTime)
	
	// 创建分片任务
	for i := int64(0); i < shardCount; i++ {
		shardID := fmt.Sprintf("shard_%d", i)
		startID := i * shardSize
		endID := startID + shardSize
		if endID > totalCount {
			endID = totalCount
		}
		
		shardKey := fmt.Sprintf("batch:task:%s:shard:%s", taskID, shardID)
		s.rdb.HSet(ctx, shardKey, map[string]interface{}{
			"shard_id":    shardID,
			"start_id":    startID,
			"end_id":      endID,
			"status":      "pending",
			"progress":    0,
			"update_time": time.Now().Format(time.RFC3339),
		})
		s.rdb.Expire(ctx, shardKey, config.ExpireTime)
		
		// 将分片加入待处理队列
		queueKey := fmt.Sprintf("batch:task:%s:queue", taskID)
		s.rdb.RPush(ctx, queueKey, fmt.Sprintf("%s:%d:%d", shardID, startID, endID))
	}
	s.rdb.Expire(ctx, fmt.Sprintf("batch:task:%s:queue", taskID), config.ExpireTime)
	
	return config, nil
}

// ProcessShard 处理单个分片
func (s *BatchUpdateService) ProcessShard(ctx context.Context, taskID string, shardID string, startID, endID int64) error {
	shardKey := fmt.Sprintf("batch:task:%s:shard:%s", taskID, shardID)
	
	// 更新分片状态为处理中
	s.rdb.HSet(ctx, shardKey, map[string]interface{}{
		"status":      "processing",
		"update_time": time.Now().Format(time.RFC3339),
	})
	
	// 批量处理数据
	currentID := startID
	for currentID < endID {
		batchEnd := currentID + int64(s.batchSize)
		if batchEnd > endID {
			batchEnd = endID
		}
		
		// 模拟批量更新操作
		// 实际项目中这里会查询数据库并更新
		err := s.processBatch(ctx, currentID, batchEnd)
		if err != nil {
			// 更新分片状态为失败
			s.rdb.HSet(ctx, shardKey, map[string]interface{}{
				"status":      "failed",
				"error":       err.Error(),
				"update_time": time.Now().Format(time.RFC3339),
			})
			return err
		}
		
		// 更新进度
		progress := ((currentID - startID) * 100) / (endID - startID)
		s.rdb.HSet(ctx, shardKey, map[string]interface{}{
			"progress":    progress,
			"update_time": time.Now().Format(time.RFC3339),
		})
		
		currentID = batchEnd
		
		// 模拟处理耗时
		time.Sleep(10 * time.Millisecond)
	}
	
	// 更新分片状态为完成
	s.rdb.HSet(ctx, shardKey, map[string]interface{}{
		"status":      "completed",
		"progress":    100,
		"update_time": time.Now().Format(time.RFC3339),
	})
	
	return nil
}

// StartWorkers 启动工作节点
func (s *BatchUpdateService) StartWorkers(ctx context.Context, taskID string) {
	var wg sync.WaitGroup
	
	for i := 0; i < s.workerCount; i++ {
		wg.Add(1)
		go func(workerID int) {
			defer wg.Done()
			s.worker(ctx, taskID, workerID)
		}(i)
	}
	
	wg.Wait()
	
	// 更新任务状态为完成
	configKey := fmt.Sprintf("batch:task:%s:config", taskID)
	s.rdb.HSet(ctx, configKey, "status", "completed")
	
	fmt.Printf("任务 %s 完成\n", taskID)
}

// worker 工作节点
func (s *BatchUpdateService) worker(ctx context.Context, taskID string, workerID int) {
	queueKey := fmt.Sprintf("batch:task:%s:queue", taskID)
	
	for {
		// 从队列获取任务
		result, err := s.rdb.BLPop(ctx, 5*time.Second, queueKey).Result()
		if err == redis.Nil {
			// 队列为空，检查是否所有分片都已完成
			if s.isTaskCompleted(ctx, taskID) {
				return
			}
			continue
		}
		if err != nil {
			fmt.Printf("Worker %d 获取任务失败: %v\n", workerID, err)
			continue
		}
		
		// 解析任务
		taskInfo := result[1]
		parts := strings.Split(taskInfo, ":")
		if len(parts) != 3 {
			continue
		}
		
		shardID := parts[0]
		startID, _ := strconv.ParseInt(parts[1], 10, 64)
		endID, _ := strconv.ParseInt(parts[2], 10, 64)
		
		fmt.Printf("Worker %d 处理分片 %s (范围: %d - %d)\n", workerID, shardID, startID, endID)
		
		// 处理分片
		err = s.ProcessShard(ctx, taskID, shardID, startID, endID)
		if err != nil {
			fmt.Printf("Worker %d 处理分片 %s 失败: %v\n", workerID, shardID, err)
		}
	}
}

// isTaskCompleted 检查任务是否完成
func (s *BatchUpdateService) isTaskCompleted(ctx context.Context, taskID string) bool {
	configKey := fmt.Sprintf("batch:task:%s:config", taskID)
	shardCount, err := s.rdb.HGet(ctx, configKey, "shard_count").Int64()
	if err != nil {
		return false
	}
	
	completedCount := int64(0)
	for i := int64(0); i < shardCount; i++ {
		shardKey := fmt.Sprintf("batch:task:%s:shard:shard_%d", taskID, i)
		status, err := s.rdb.HGet(ctx, shardKey, "status").Result()
		if err == nil && (status == "completed" || status == "failed") {
			completedCount++
		}
	}
	
	return completedCount == shardCount
}

// GetTaskProgress 获取任务进度
func (s *BatchUpdateService) GetTaskProgress(ctx context.Context, taskID string) (float64, error) {
	configKey := fmt.Sprintf("batch:task:%s:config", taskID)
	shardCount, err := s.rdb.HGet(ctx, configKey, "shard_count").Int64()
	if err != nil {
		return 0, err
	}
	
	totalProgress := int64(0)
	for i := int64(0); i < shardCount; i++ {
		shardKey := fmt.Sprintf("batch:task:%s:shard:shard_%d", taskID, i)
		progress, err := s.rdb.HGet(ctx, shardKey, "progress").Int64()
		if err != nil {
			progress = 0
		}
		totalProgress += progress
	}
	
	return float64(totalProgress) / float64(shardCount), nil
}

// 使用示例
func main() {
	ctx := context.Background()
	
	// 创建Redis客户端
	rdb := redis.NewClient(&redis.Options{
		Addr: "localhost:6379",
	})
	
	// 创建批量更新服务
	service := NewBatchUpdateService(rdb, 5, 1000)
	
	// 创建任务
	taskID := "update_product_20240101"
	totalCount := int64(1000000) // 100万条数据
	
	config, err := service.CreateTask(ctx, taskID, totalCount)
	if err != nil {
		fmt.Printf("创建任务失败: %v\n", err)
		return
	}
	
	fmt.Printf("任务创建成功: %+v\n", config)
	
	// 启动工作节点
	go service.StartWorkers(ctx, taskID)
	
	// 监控任务进度
	ticker := time.NewTicker(5 * time.Second)
	defer ticker.Stop()
	
	for range ticker.C {
		progress, err := service.GetTaskProgress(ctx, taskID)
		if err != nil {
			fmt.Printf("获取进度失败: %v\n", err)
			continue
		}
		
		fmt.Printf("任务进度: %.2f%%\n", progress)
		
		if progress >= 100 {
			break
		}
	}
}
