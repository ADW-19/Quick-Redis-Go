## 8.6 验证码系统

### 场景故事：防止恶意注册和暴力破解

**故事背景**：
某社交平台在上线初期，遭遇了大量的恶意注册和暴力破解攻击。攻击者使用自动化脚本批量注册账号，尝试暴力破解用户密码，导致系统资源被占用，正常用户无法使用服务。技术团队需要实现一个验证码系统来防止这些攻击。

**业务需求**：
1. 防止自动化脚本批量注册
2. 防止暴力破解密码
3. 确保验证码的安全性和有效性
4. 提供良好的用户体验，验证码易于识别
5. 防止验证码被恶意刷取

### 应用场景

- 用户注册验证码：防止批量注册
- 登录验证码：防止暴力破解
- 密码重置验证码：确保操作安全性
- 支付验证码：保护用户资金安全

### 方案对比

**方案一：基于数据库的验证码存储**
- **优点**：
  - 数据持久化，可靠性高
  - 支持复杂查询和分析
  - 事务支持，数据一致性好
- **缺点**：
  - 性能差，数据库成为瓶颈
  - 并发能力有限
  - 验证码过期管理复杂
  - 不适合高并发场景

**方案二：基于内存的验证码存储**
- **优点**：
  - 性能高，内存操作
  - 实时性好，无延迟
- **缺点**：
  - 数据易丢失，服务器重启后数据清零
  - 不支持分布式环境
  - 内存容量有限
  - 无法实现跨服务器验证

**方案三：基于Redis的验证码存储**
- **优点**：
  - 高性能，支持高并发
  - 支持过期时间，自动清理
  - 数据持久化选项（RDB/AOF）
  - 支持分布式部署
  - 实现简单，API友好
  - 支持频率限制，防止刷取
- **缺点**：
  - 需要额外的Redis服务器资源
  - 网络传输开销
  - 数据可能存在短暂不一致（在主从复制场景下）

**选择理由**：
考虑到社交平台的高并发需求和防刷要求，选择基于Redis的验证码存储方案。Redis的高性能特性能够应对大量的验证码生成和验证请求，支持过期时间自动清理验证码，同时通过原子操作实现频率限制，有效防止验证码被恶意刷取。相比其他方案，Redis验证码系统在性能、可靠性和功能方面都具有明显优势。

### 代码实现

```go
package main

import (
	"context"
	"fmt"
	"math/rand"
	"time"

	"github.com/go-redis/redis/v8"
)

// VerificationService 验证码服务
type VerificationService struct {
	rdb *redis.Client
}

// VerificationConfig 验证码配置
type VerificationConfig struct {
	CodeLength      int           // 验证码长度
	ExpireTime      time.Duration // 验证码过期时间
	IPLimitCount    int           // IP限制次数
	IPLimitWindow   time.Duration // IP限制时间窗口
	PhoneLimitCount int           // 手机号限制次数
	PhoneLimitWindow time.Duration // 手机号限制时间窗口
}

var defaultConfig = VerificationConfig{
	CodeLength:       6,
	ExpireTime:       5 * time.Minute,
	IPLimitCount:     10,
	IPLimitWindow:    time.Minute,
	PhoneLimitCount:  5,
	PhoneLimitWindow: time.Minute,
}

// NewVerificationService 创建验证码服务
func NewVerificationService(rdb *redis.Client) *VerificationService {
	return &VerificationService{rdb: rdb}
}

// GenerateCode 生成验证码
func (s *VerificationService) GenerateCode() string {
	code := rand.Intn(900000) + 100000 // 生成6位数字
	return fmt.Sprintf("%06d", code)
}

// SendVerificationCode 发送验证码
func (s *VerificationService) SendVerificationCode(ctx context.Context, phone string, ip string) error {
	// 检查IP频率限制
	ipKey := fmt.Sprintf("verify:limit:ip:%s", ip)
	ipCount, err := s.rdb.Incr(ctx, ipKey).Result()
	if err != nil {
		return fmt.Errorf("检查IP频率失败: %w", err)
	}
	
	// 设置IP限制窗口
	if ipCount == 1 {
		s.rdb.Expire(ctx, ipKey, defaultConfig.IPLimitWindow)
	}
	
	if ipCount > int64(defaultConfig.IPLimitCount) {
		return fmt.Errorf("该IP请求过于频繁，请稍后再试")
	}
	
	// 检查手机号频率限制
	phoneKey := fmt.Sprintf("verify:limit:phone:%s", phone)
	phoneCount, err := s.rdb.Incr(ctx, phoneKey).Result()
	if err != nil {
		return fmt.Errorf("检查手机号频率失败: %w", err)
	}
	
	// 设置手机号限制窗口
	if phoneCount == 1 {
		s.rdb.Expire(ctx, phoneKey, defaultConfig.PhoneLimitWindow)
	}
	
	if phoneCount > int64(defaultConfig.PhoneLimitCount) {
		return fmt.Errorf("该手机号请求过于频繁，请稍后再试")
	}
	
	// 生成验证码
	code := s.GenerateCode()
	
	// 存储验证码到Redis
	codeKey := fmt.Sprintf("verify:code:%s", phone)
	err = s.rdb.Set(ctx, codeKey, code, defaultConfig.ExpireTime).Err()
	if err != nil {
		return fmt.Errorf("存储验证码失败: %w", err)
	}
	
	// 发送验证码（这里模拟发送，实际项目中调用短信网关）
	fmt.Printf("向手机号 %s 发送验证码: %s\n", phone, code)
	
	return nil
}

// VerifyCode 验证验证码
func (s *VerificationService) VerifyCode(ctx context.Context, phone string, code string) (bool, error) {
	codeKey := fmt.Sprintf("verify:code:%s", phone)
	
	// 获取存储的验证码
	storedCode, err := s.rdb.Get(ctx, codeKey).Result()
	if err == redis.Nil {
		return false, fmt.Errorf("验证码已过期或不存在")
	}
	if err != nil {
		return false, fmt.Errorf("获取验证码失败: %w", err)
	}
	
	// 对比验证码
	if storedCode != code {
		return false, nil
	}
	
	// 验证成功，删除验证码（防止重用）
	s.rdb.Del(ctx, codeKey)
	
	return true, nil
}

// GetRemainingTime 获取验证码剩余有效时间
func (s *VerificationService) GetRemainingTime(ctx context.Context, phone string) (time.Duration, error) {
	codeKey := fmt.Sprintf("verify:code:%s", phone)
	ttl, err := s.rdb.TTL(ctx, codeKey).Result()
	if err != nil {
		return 0, err
	}
	return ttl, nil
}

// 使用示例
func main() {
	ctx := context.Background()
	
	// 创建Redis客户端
	rdb := redis.NewClient(&redis.Options{
		Addr: "localhost:6379",
	})
	
	// 创建验证码服务
	service := NewVerificationService(rdb)
	
	phone := "13800138000"
	ip := "192.168.1.100"
	
	// 发送验证码
	err := service.SendVerificationCode(ctx, phone, ip)
	if err != nil {
		fmt.Printf("发送验证码失败: %v\n", err)
		return
	}
	
	// 模拟用户输入验证码
	var inputCode string
	fmt.Print("请输入验证码: ")
	fmt.Scan(&inputCode)
	
	// 验证验证码
	valid, err := service.VerifyCode(ctx, phone, inputCode)
	if err != nil {
		fmt.Printf("验证失败: %v\n", err)
		return
	}
	
	if valid {
		fmt.Println("验证码验证成功！")
	} else {
		fmt.Println("验证码验证失败！")
	}
}
```
