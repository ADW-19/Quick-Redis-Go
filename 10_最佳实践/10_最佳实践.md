## 10.1 键名设计

### 设计原则

- **使用统一的命名规范**：如 `业务:对象:ID:属性`。
- **避免使用过长的键名**：键名会占用内存，过长的键名会浪费内存。
- **使用哈希结构存储对象**：对于对象，使用哈希结构可以节省内存。

### 键名设计规范图

```
推荐的键名设计规范：
+----------------+----------------+----------------+----------------+
|   业务模块     |   对象类型     |   对象ID       |   属性         |
+----------------+----------------+----------------+----------------+
|   user         |   profile      |   12345        |   name         |
+----------------+----------------+----------------+----------------+
|   order        |   info         |   67890        |   status       |
+----------------+----------------+----------------+----------------+
|   product      |   detail       |   11111        |   price        |
+----------------+----------------+----------------+----------------+

完整键名示例：
user:profile:12345:name
order:info:67890:status
product:detail:11111:price

不推荐的键名设计：
+----------------+
|   u_p_12345_n  |
+----------------+
|   o_i_67890_s  |
+----------------+
|   p_d_11111_p  |
+----------------+
问题：缩写不清晰，难以理解和维护
```

### 数据结构选择对比图

```
字符串存储对象：
+----------------+----------------+----------------+----------------+
|   user:12345   |   user:12345   |   user:12345   |   user:12345   |
|   :name        |   :age         |   :gender      |   :email       |
+----------------+----------------+----------------+----------------+
|   "张三"       |   "18"         |   "男"         |   "zhang@..."   |
+----------------+----------------+----------------+----------------+
内存占用：4个键，每个键都有额外开销

哈希存储对象：
+----------------+
|   user:12345   |
+----------------+
|   name: "张三"  |
|   age: "18"     |
|   gender: "男"   |
|   email: "zhang@..." |
+----------------+
内存占用：1个键，节省内存开销
```

## 10.2 数据过期策略

### 策略说明

- **设置合理的过期时间**：根据数据的时效性设置合理的过期时间。
- **使用惰性删除和定期删除**：Redis会定期删除过期的键，同时在访问键时会检查键是否过期。
- **避免大量键同时过期**：设置随机的过期时间，避免缓存雪崩。

// 设置随机过期时间，防止缓存雪崩
func setCacheWithRandomExpire(ctx context.Context, rdb *redis.Client, key string, value interface{}, baseExpire time.Duration) error {
    // 在基础过期时间上增加随机时间（0-300秒）
    randomExpire := baseExpire + time.Duration(rand.Intn(300))*time.Second
    
    err := rdb.Set(ctx, key, value, randomExpire).Err()
    if err != nil {
        return err
    }
    
    return nil
}

// Redis监控指标收集
type RedisMonitor struct {
    rdb *redis.Client
}

// 获取Redis监控指标
func (m *RedisMonitor) GetMetrics(ctx context.Context) (*RedisMetrics, error) {
    info, err := m.rdb.Info(ctx).Result()
    if err != nil {
        return nil, err
    }
    
    // 解析INFO命令的输出
    metrics := &RedisMetrics{
        MemoryUsage:    m.parseMemoryUsage(info),
        CPUUsage:       m.parseCPUUsage(info),
        ConnectionCount: m.parseConnectionCount(info),
        SlowLogCount:   m.getSlowLogCount(ctx),
    }
    
    return metrics, nil
}

// 检查指标是否超过阈值
func (m *RedisMonitor) CheckThresholds(metrics *RedisMetrics) []Alert {
    var alerts []Alert
    
    // 检查内存使用率
    if metrics.MemoryUsage > 0.9 {
        alerts = append(alerts, Alert{
            Level:   "critical",
            Metric:  "memory_usage",
            Value:   metrics.MemoryUsage,
            Message: "内存使用率超过90%",
        })
    } else if metrics.MemoryUsage > 0.8 {
        alerts = append(alerts, Alert{
            Level:   "warning",
            Metric:  "memory_usage",
            Value:   metrics.MemoryUsage,
            Message: "内存使用率超过80%",
        })
    }
    
    // 检查CPU使用率
    if metrics.CPUUsage > 0.85 {
        alerts = append(alerts, Alert{
            Level:   "critical",
            Metric:  "cpu_usage",
            Value:   metrics.CPUUsage,
            Message: "CPU使用率超过85%",
        })
    } else if metrics.CPUUsage > 0.7 {
        alerts = append(alerts, Alert{
            Level:   "warning",
            Metric:  "cpu_usage",
            Value:   metrics.CPUUsage,
            Message: "CPU使用率超过70%",
        })
    }
    
    return alerts
}

// 定期监控
func (m *RedisMonitor) StartMonitoring(ctx context.Context, interval time.Duration) {
    ticker := time.NewTicker(interval)
    defer ticker.Stop()
    
    for {
        select {
        case <-ctx.Done():
            return
        case <-ticker.C:
            metrics, err := m.GetMetrics(ctx)
            if err != nil {
                fmt.Printf("获取监控指标失败: %v\n", err)
                continue
            }
            
            alerts := m.CheckThresholds(metrics)
            for _, alert := range alerts {
                m.sendAlert(alert)
            }
        }
    }
}

// 发送告警

    fmt.Printf("告警: %s - %s (值: %.2f)\n", alert.Level, alert.Message, alert.Value)
    // 这里可以集成邮件、短信、钉钉等告警方式

## 10.4 安全措施

### 安全策略

- **设置密码**：使用 `requirepass` 配置项设置密码。
- **限制访问IP**：使用 `bind` 配置项限制只能从指定IP访问。
- **使用ACL**：Redis 6.0引入了ACL（Access Control List），可以更细粒度地控制用户权限。
- **避免使用危险命令**：如 `FLUSHALL`、`FLUSHDB` 等，可以通过 `rename-command` 配置项禁用这些命令。

// Redis安全配置
type RedisSecurityConfig struct {
    Password     string
    ACLUsers     []ACLUser
    DisabledCmds []string
}

// ACL用户配置
type ACLUser struct {
    Username    string
    Password    string
    Commands    string
    Keys        string
    Databases   string
}

// 创建安全的Redis客户端
func NewSecureRedisClient(config RedisSecurityConfig) (*redis.Client, error) {
    rdb := redis.NewClient(&redis.Options{
        Addr:     "localhost:6379",
        Password: config.Password,
        DB:       0,
    })
    
    // 测试连接
    ctx := context.Background()
    _, err := rdb.Ping(ctx).Result()
    if err != nil {
        return nil, err
    }
    
    // 配置ACL用户
    for _, user := range config.ACLUsers {
        aclCmd := fmt.Sprintf("ACL SETUSER %s on >%s %s %s %s",
            user.Username,
            user.Password,
            user.Commands,
            user.Keys,
            user.Databases,
        )
        
        err := rdb.Do(ctx, "ACL", strings.Split(aclCmd, " ")...").Err()
        if err != nil {
            return nil, fmt.Errorf("配置ACL用户失败: %w", err)
        }
    }
    
    // 禁用危险命令
    for _, cmd := range config.DisabledCmds {
        err := rdb.Do(ctx, "RENAME-COMMAND", cmd, "").Err()
        if err != nil {
            return nil, fmt.Errorf("禁用命令失败: %w", err)
        }
    }
    
    return rdb, nil
}

// 使用示例
func main() {
    config := RedisSecurityConfig{
        Password: "your_secure_password",
        ACLUsers: []ACLUser{
            {
                Username:  "admin",
                Password:  "admin_password",
                Commands: "+@all",
                Keys:      "*",
                Databases: "*",
            },
            {
                Username:  "readonly",
                Password:  "readonly_password",
                Commands: "+@read",
                Keys:      "*",
                Databases: "*",
            },
            {
                Username:  "app_user",
                Password:  "app_password",
                Commands: "+get +set +del +hget +hset",
                Keys:      "app:*",
                Databases: "0-9",
            },
        },
        DisabledCmds: []string{
            "FLUSHALL",
            "FLUSHDB",
            "CONFIG",
            "SHUTDOWN",
        },
    }
    
    rdb, err := NewSecureRedisClient(config)
    if err != nil {
        fmt.Printf("创建Redis客户端失败: %v\n", err)
        return
    }
    
    defer rdb.Close()
    
    fmt.Println("Redis客户端创建成功，已配置安全措施")
}

func (m *RedisBackupManager) CreateRDBBackup(ctx context.Context) (string, error) {
    timestamp := time.Now().Format("20060102_150405")
    backupFile := fmt.Sprintf("%s/redis_backup_%s.rdb", m.backupDir, timestamp)
    
    // 执行BGSAVE命令生成RDB文件
    err := m.rdb.BgSave(ctx).Err()
    if err != nil {
        return "", fmt.Errorf("创建RDB备份失败: %w", err)
    }
    
    // 等待BGSAVE完成
    for {
        info, err := m.rdb.Info(ctx, "persistence").Result()
        if err != nil {
            return "", fmt.Errorf("获取持久化信息失败: %w", err)
        }
        
        if strings.Contains(info, "rdb_bgsave_in_progress:0") {
            break
        }
        
        time.Sleep(1 * time.Second)
    }
    
    // 复制RDB文件到备份目录
    rdbPath := "/var/lib/redis/dump.rdb"
    err = copyFile(rdbPath, backupFile)
    if err != nil {
        return "", fmt.Errorf("复制RDB文件失败: %w", err)
    }
    
    // 上传到远程备份
    if m.remoteBackup != nil {
        err = m.remoteBackup.Upload(ctx, backupFile)
        if err != nil {
            fmt.Printf("上传远程备份失败: %v\n", err)
        }
    }
    
    return backupFile, nil
}

// 从RDB备份恢复
func (m *RedisBackupManager) RestoreFromRDB(ctx context.Context, backupFile string) error {
    // 停止Redis服务
    err := stopRedisService()
    if err != nil {
        return fmt.Errorf("停止Redis服务失败: %w", err)
    }
    
    // 备份当前的RDB文件
    rdbPath := "/var/lib/redis/dump.rdb"
    err = copyFile(rdbPath, rdbPath+".bak")
    if err != nil {
        return fmt.Errorf("备份当前RDB文件失败: %w", err)
    }
    
    // 复制备份文件到Redis目录
    err = copyFile(backupFile, rdbPath)
    if err != nil {
        return fmt.Errorf("复制备份文件失败: %w", err)
    }
    
    // 启动Redis服务
    err = startRedisService()
    if err != nil {
        return fmt.Errorf("启动Redis服务失败: %w", err)
    }
    
    // 验证恢复结果
    err = m.rdb.Ping(ctx).Err()
    if err != nil {
        return fmt.Errorf("验证Redis恢复失败: %w", err)
    }
    
    return nil
}

// 定期备份
func (m *RedisBackupManager) StartScheduledBackup(ctx context.Context) {
    ticker := time.NewTicker(m.backupSchedule)
    defer ticker.Stop()
    
    for {
        select {
        case <-ctx.Done():
            return
        case <-ticker.C:
            backupFile, err := m.CreateRDBBackup(ctx)
            if err != nil {
                fmt.Printf("创建备份失败: %v\n", err)
                continue
            }
            
            fmt.Printf("备份创建成功: %s\n", backupFile)
            
            // 清理旧备份（保留最近7天）
            m.cleanOldBackups(7 * 24 * time.Hour)
        }
    }
}

// 清理旧备份
func (m *RedisBackupManager) cleanOldBackups(maxAge time.Duration) {
    files, err := os.ReadDir(m.backupDir)
    if err != nil {
        fmt.Printf("读取备份目录失败: %v\n", err)
        return
    }
    
    now := time.Now()
    for _, file := range files {
        if file.IsDir() {
            continue
        }
        
        info, err := file.Info()
        if err != nil {
            continue
        }
        
        if now.Sub(info.ModTime()) > maxAge {
            filePath := filepath.Join(m.backupDir, file.Name())
            err := os.Remove(filePath)
            if err != nil {
                fmt.Printf("删除旧备份失败: %v\n", err)
            } else {
                fmt.Printf("删除旧备份: %s\n", filePath)
            }
        }
    }
}

// 测试恢复流程
func (m *RedisBackupManager) TestRestore(ctx context.Context) error {
    // 创建测试备份
    backupFile, err := m.CreateRDBBackup(ctx)
    if err != nil {
        return fmt.Errorf("创建测试备份失败: %w", err)
    }
    
    // 获取当前数据快照
    keysBefore, err := m.rdb.Keys(ctx, "*").Result()
    if err != nil {
        return fmt.Errorf("获取当前数据失败: %w", err)
    }
    
    // 恢复备份
    err = m.RestoreFromRDB(ctx, backupFile)
    if err != nil {
        return fmt.Errorf("恢复备份失败: %w", err)
    }
    
    // 验证恢复结果
    keysAfter, err := m.rdb.Keys(ctx, "*").Result()
    if err != nil {
        return fmt.Errorf("获取恢复后数据失败: %w", err)
    }
    
    if len(keysBefore) != len(keysAfter) {
        return fmt.Errorf("恢复验证失败: 键数量不一致")
    }
    
    fmt.Println("恢复测试成功")
    return nil
}

// 辅助函数：复制文件
func copyFile(src, dst string) error {
    source, err := os.Open(src)
    if err != nil {
        return err
    }
    defer source.Close()
    
    destination, err := os.Create(dst)
    if err != nil {
        return err
    }
    defer destination.Close()
    
    _, err = io.Copy(destination, source)
    return err
}
